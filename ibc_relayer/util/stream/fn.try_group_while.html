Example

use ibc_relayer::util::stream::try_group_while;
use futures::{stream, StreamExt, executor::block_on};

let input = stream::iter(vec![
    Ok(1),
    Ok(1),
    Ok(2),
    Err(()),
    Ok(2),
    Ok(2),
    Ok(3),
    Ok(3),
    Err(()),
]);

let output = try_group_while(input, |a, b| a == b).collect::<Vec<_>>();

assert_eq!(
    block_on(output),
    vec![
        Ok(vec![1, 1]),
        Ok(vec![2]),
        Err(()),
        Ok(vec![2, 2]),
        Ok(vec![3]),
        Ok(vec![3]),
        Err(()),
    ]
);
